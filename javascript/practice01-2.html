1. 구구단 출력하기

1.1 for문을 이용해서 1단부터 9단까지 출력하시오!

{
    for (let i = 2; i<=9; i++){
        document.write("<br>")
        
        for (let j = 1; j<=9; j++){
            let sum = i * j;
            document.write(i + " X " + j + " = " + sum, "<br>")
        }
    }
}

1.2 while문을 이용해서 1단부터 9단까지 출력하시오!

{
    let i = 2
    while(i <= 9){
        document.write("<br>")

        let j = 1
        while(j <= 9){
            let sum = i * j;
            document.write(i + " X " + j + " = " + sum, "<br>")
            j++ 
        };
        i++
    }
}
